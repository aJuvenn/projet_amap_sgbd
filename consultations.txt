FROM souscrire_a
WHERE id_contrat=25

2.
SELECT id_livraison
FROM Livraison
WHERE date_livraison >= "01/12"
AND date_livraison <= "31/12"

3.
SELECT id_livraison
FROM livraison
WHERE id_foyer IS NULL

4.
SELECT sa.id_contrat,date_livraison
FROM souscrire_a sa
JOIN prevision_calendrier
USING id_contrat
JOIN livraison
USING id_livraison
WHERE id_foyer=42
ORDER BY date_livraison
